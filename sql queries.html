SQL queries

show all employees with names, position and salary

SELECT employees.first_name, employees.last_name, employees.position, salaries.salary_value FROM salaries INNER JOIN employees ON employees.emp_ID

show all vehicles and all their data

SELECT vehicle.VIN, vehicle.license_plate, vehicle.gas, vehicle.reg_date, vehicle.max_passenger, vehicle.cond, vehicle.km, vehicle.class, vehicle.car_model, vehicle.manufacturer FROM vehicle

show employees and their address

SELECT employees.first_name, employees.last_name, employees.position, address.address_ID, address.street, address.house_number, address.zip_code, address.city, address.country FROM address INNER JOIN employees ON address.address_ID
